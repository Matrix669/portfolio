import { NextRequest, NextResponse } from 'next/server'
import Stripe from 'stripe'
import { redis } from '@/lib/redis'
import { baseUrlFront } from '@/lib/getBaseUrlFront'

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!)

interface MagicLinkData {
	customerId: string
	email: string
	createdAt: number
}

export async function GET(req: NextRequest) {
	const token = req.nextUrl.searchParams.get('token')

	console.log(`🔍 Weryfikacja tokenu: ${token}`)

	if (!token) {
		console.log('❌ Brak tokenu w URL')
		return NextResponse.redirect(
			`${baseUrlFront}/subscription?error=invalid-token`
		)
	}

	try {
		// Pobierz dane z Redis
		const data = await redis.get<MagicLinkData>(`magic:${token}`)

		if (!data) {
			console.log('❌ Token nie istnieje lub wygasł')
			return NextResponse.redirect(
				`${baseUrlFront}/subscription?error=expired-token`
			)
		}

		console.log(`✅ Token znaleziony dla customera: ${data.customerId}`)

		// Usuń token z Redis (jednorazowe użycie)
		await redis.del(`magic:${token}`)
		console.log(`🗑️ Token usunięty z Redis`)

		// Wygeneruj Stripe Customer Portal Session
		const portalSession = await stripe.billingPortal.sessions.create({
			customer: data.customerId,
			return_url: `${baseUrlFront}/wesprzyj-nas`,
		})

		console.log(`✅ Portal session utworzony`)
		console.log(`🔗 Przekierowuję na: ${portalSession.url}`)

		// Przekieruj użytkownika na Stripe Portal
		return NextResponse.redirect(portalSession.url)
	} catch (error) {
		console.error('❌ Błąd weryfikacji tokenu:', error)
		return NextResponse.redirect(
			`${baseUrlFront}/subscription?error=portal-error`
		)
	}
}
