import { NextRequest, NextResponse } from 'next/server'

// Przykład integracji z API InPost
// Dokumentacja: https://dokumentacja-inpost.atlassian.net/wiki/spaces/PL/overview

interface InPostShipmentData {
	receiver: {
		email: string
		phone: string
		name: string
	}
	parcels: Array<{
		template: string
		dimensions: {
			length: string
			width: string
			height: string
		}
		weight: {
			amount: string
			unit: 'kg'
		}
	}>
	custom_attributes: {
		target_point?: string // dla paczkomatów
		sending_method: 'parcel_locker' | 'courier'
	}
	service: 'inpost_locker_standard' | 'inpost_courier_standard'
}

export async function POST(request: NextRequest) {
	try {
		const { orderData, shippingMethod } = await request.json()

		// Konfiguracja API InPost
		const INPOST_API_URL = process.env.INPOST_API_URL || 'https://api-shipx-pl.easypack24.net'
		const INPOST_API_TOKEN = process.env.INPOST_API_TOKEN

		if (!INPOST_API_TOKEN) {
			throw new Error('Brak tokena API InPost')
		}

		// Przygotowanie danych przesyłki
		const shipmentData: InPostShipmentData = {
			receiver: {
				email: orderData.email,
				phone: orderData.phoneNumber,
				name: `${orderData.firstName} ${orderData.lastName}`,
			},
			parcels: [
				{
					template: 'small', // small, medium, large, avizo
					dimensions: {
						length: '20',
						width: '15',
						height: '10',
					},
					weight: {
						amount: '0.5',
						unit: 'kg',
					},
				},
			],
			custom_attributes: {
				sending_method: shippingMethod === 'Inpost Paczkomat' ? 'parcel_locker' : 'courier',
				...(shippingMethod === 'Inpost Paczkomat' && {
					target_point: orderData.selectedParcelLocker, // ID wybranego paczkomatu
				}),
			},
			service: shippingMethod === 'Inpost Paczkomat' 
				? 'inpost_locker_standard' 
				: 'inpost_courier_standard',
		}

		// Wywołanie API InPost
		const response = await fetch(`${INPOST_API_URL}/v1/organizations/{organization_id}/shipments`, {
			method: 'POST',
			headers: {
				'Authorization': `Bearer ${INPOST_API_TOKEN}`,
				'Content-Type': 'application/json',
			},
			body: JSON.stringify(shipmentData),
		})

		if (!response.ok) {
			const errorData = await response.json()
			throw new Error(`InPost API Error: ${errorData.message}`)
		}

		const shipmentResult = await response.json()

		return NextResponse.json({
			success: true,
			shipmentId: shipmentResult.id,
			trackingNumber: shipmentResult.tracking_number,
			labelUrl: shipmentResult.label_url,
		})
	} catch (error) {
		console.error('Błąd podczas tworzenia przesyłki InPost:', error)
		return NextResponse.json(
			{ error: 'Błąd podczas tworzenia przesyłki' },
			{ status: 500 }
		)
	}
}

// Endpoint do pobierania listy paczkomatów
export async function GET(request: NextRequest) {
	try {
		const { searchParams } = new URL(request.url)
		const city = searchParams.get('city')
		const zipCode = searchParams.get('zipCode')

		const INPOST_API_URL = process.env.INPOST_API_URL || 'https://api-shipx-pl.easypack24.net'

		// Pobieranie listy paczkomatów
		const response = await fetch(
			`${INPOST_API_URL}/v1/points?type=parcel_locker&city=${city}&post_code=${zipCode}`,
			{
				method: 'GET',
				headers: {
					'Content-Type': 'application/json',
				},
			}
		)

		if (!response.ok) {
			throw new Error('Błąd podczas pobierania listy paczkomatów')
		}

		const parcelLockers = await response.json()

		return NextResponse.json({
			success: true,
			parcelLockers: parcelLockers.items || [],
		})
	} catch (error) {
		console.error('Błąd podczas pobierania paczkomatów:', error)
		return NextResponse.json(
			{ error: 'Błąd podczas pobierania paczkomatów' },
			{ status: 500 }
		)
	}
}
