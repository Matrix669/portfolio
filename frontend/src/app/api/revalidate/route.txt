import { revalidatePath } from 'next/cache'
import { NextRequest, NextResponse } from 'next/server'

// Typy dla danych z body żądania
interface RevalidateBody {
	model: string
	// slug?: string
	// category?: string
	// infoPage?: string
	entry?: {
		slug?: string
		category?: string
		infoPage?: string
	}
}

export async function POST(req: NextRequest) {
	// Sprawdzenie tokena
	const token = req.nextUrl.searchParams.get('token')
	if (token !== process.env.REVALIDATE_TOKEN) {
		return NextResponse.json({ message: 'Nieprawidłowy token' }, { status: 401 })
	}

	try {
		// Pobranie modelu i sluga z body żądania
		// const { model, slug, category, infoPage }: RevalidateBody = await req.json()
		const { model, entry }: RevalidateBody = await req.json()
		const slug = entry?.slug
		const category = entry?.category
		const infoPage = entry?.infoPage

		console.log('Otrzymano dane webhooka:', { model, slug, category, infoPage })

		let pathToRevalidate: string[] = []

		// Obsługa różnych modeli
		switch (model) {
			case 'strona-glowna':
				pathToRevalidate = ['/']
				break
			case 'wesprzyj-nas':
				pathToRevalidate = ['/wesprzyj-nas']
				break
			case 'kontakt':
				pathToRevalidate = ['/kontakt']
				break
			case 'jak-mozesz-pomoc':
				pathToRevalidate = ['/jak-mozesz-pomoc']
				break
			case 'nasza-historia':
				pathToRevalidate = ['/nasza-historia']
				break
			case 'nasz-zespol':
				pathToRevalidate = ['/nasz-zespol']
				break
			case 'nasza-misja-cele':
				pathToRevalidate = ['/nasza-misja-cele']
				break
			case 'nasze-grupy':
				pathToRevalidate = ['/nasze-grupy']
				break
			case 'sprawozdania':
				pathToRevalidate = ['/informacje/sprawozdania']
				break
			case 'post-projekt':
				if (category && ['obecne', 'zakonczone'].includes(category)) {
					pathToRevalidate = []
					if (slug) {
						pathToRevalidate.push(`/projekty-${category}/${slug}`)
					}
					pathToRevalidate.push('/projekty-obecne') // Rewalidacja obu kategorii
					pathToRevalidate.push('/projekty-zakonczone')
				} else {
					console.warn(`Nieprawidłowa kategoria: ${category}`)
				}
				break
			case 'projekty':
				pathToRevalidate = ['/projekty-obecne', '/projekty-zakonczone']
				break
			case 'informacje':
				pathToRevalidate = infoPage ? [`/informacje/${infoPage}`] : ['/informacje']
				break
			default:
				break
		}

		// Logowanie ścieżek do rewalidacji
		console.log(`Revalidating: ${pathToRevalidate.join(', ')}`)

		// Rewalidacja wszystkich ścieżek
		await Promise.all(pathToRevalidate.map(path => revalidatePath(path)))

		return NextResponse.json({ revalidated: true, paths: pathToRevalidate }, { status: 200 })
	} catch (err) {
		console.error('Błąd podczas rewalidacji:', err)
		return NextResponse.json({ message: 'Błąd rewalidacji', error: (err as Error).message }, { status: 500 })
	}
}
