import { NextRequest, NextResponse } from 'next/server'
import Stripe from 'stripe'

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!)

export async function POST(req: NextRequest) {
	const { sessionId } = await req.json()

	if (!sessionId) {
		return NextResponse.json({ error: 'Brak ID sesji' }, { status: 400 })
	}

	try {
		const session = await stripe.checkout.sessions.retrieve(sessionId)
		// console.log('Status anulowanej sesji:', session.status)

		if (session.status === 'open') {
			const canceledSession = await stripe.checkout.sessions.expire(sessionId)
			const response = {
				status: 'open',
				message: 'Płatność została anulowana',
				session: canceledSession,
			}
			// console.log('API Response (open):', response)
			return NextResponse.json(response)
		}
		// Jeśli płatność jest już zakończona, nie możemy jej anulować
		if (session.status === 'complete') {
			const response = {
				status: 'complete',
				message: 'Płatność została już zrealizowana',
			}
			// console.log('API Response (complete):', response)
			return NextResponse.json(response)
		}

		// Jeśli sesja jest już anulowana, nie możemy jej ponownie anulować
		if (session.status === 'expired') {
			const response = {
				status: 'expired',
				message: 'Sesja została już anulowana',
			}
			// console.log('API Response (expired):', response)
			return NextResponse.json(response)
		}

		return NextResponse.json(
			{
				status: 'error',
				message: `Nieobsługiwany status sesji: ${session.status}`,
			},
			{ status: 400 }
		)
	} catch (error) {
		console.error('Błąd podczas pobierania sesji:', error)
		return NextResponse.json(
			{
				error: 'Wystąpił błąd podczas anulowania płatności',
				details: (error as Error).message,
			},
			{ status: 500 }
		)
	}
}
