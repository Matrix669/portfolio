import { NextRequest, NextResponse } from 'next/server'
import Stripe from 'stripe'

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!)

export async function POST(req: NextRequest) {
	const { email, amount, group, monthlySupport, paymentMethod } = (await req.json()) as {
		email: string
		amount: number
		group: string
		monthlySupport: boolean
		paymentMethod: string
	}

	let paymentMethodTypes: Stripe.Checkout.SessionCreateParams.PaymentMethodType[] = []
	if (paymentMethod === 'Google Pay' || paymentMethod === 'Apple Pay') {
		paymentMethodTypes = ['card']
	} else if (paymentMethod === 'Paypal') {
		paymentMethodTypes = ['paypal']
	} else if (paymentMethod === 'Przelewy24') {
		paymentMethodTypes = ['p24']
	}

	try {
		// ✅ POPRAWKA: Sprawdź czy customer już istnieje
		const emailNormalized = email.toLowerCase().trim()
		let customerId: string

		const existingCustomers = await stripe.customers.list({
			email: emailNormalized,
			limit: 1,
		})

		if (existingCustomers.data.length > 0) {
			// Użyj istniejącego customera
			customerId = existingCustomers.data[0].id
			console.log(`✅ Używam istniejącego customera: ${customerId} dla ${emailNormalized}`)
		} else {
			// Utwórz nowego customera tylko jeśli nie istnieje
			const newCustomer = await stripe.customers.create({
				email: emailNormalized,
				metadata: {
					source: 'fundacja_drachma_website',
					created_via: 'checkout_form',
				},
			})
			customerId = newCustomer.id
			console.log(`🆕 Utworzono nowego customera: ${customerId} dla ${emailNormalized}`)
		}

		let session: Stripe.Checkout.Session

		if (monthlySupport) {
			session = await stripe.checkout.sessions.create({
				customer: customerId, // ✅ Użyj customerId zamiast customer.id
				mode: 'subscription',
				payment_method_types: paymentMethodTypes,
				payment_method_options: {
					card: {
						request_three_d_secure: 'automatic',
					},
				},
				subscription_data: {
					description: `Miesięczne wsparcie dla grupy "${group}".`,
					metadata: {
						group: group,
						support_type: 'monthly',
					},
				},
				line_items: [
					{
						price_data: {
							currency: 'pln',
							product_data: {
								name: `"${group}"`,
								description: `Miesięczne wsparcie dla grupy "${group}".`,
							},
							unit_amount: amount,
							recurring: {
								interval: 'month',
							},
						},
						quantity: 1,
					},
				],
				ui_mode: 'embedded',
				return_url: `${req.headers.get('origin')}/paymentResult?session_id={CHECKOUT_SESSION_ID}`,
			})
		} else {
			session = await stripe.checkout.sessions.create({
				customer: customerId, // ✅ Użyj customerId zamiast customer.id
				mode: 'payment',
				payment_method_types: paymentMethodTypes,
				payment_method_options: {
					card: {
						request_three_d_secure: 'automatic',
					},
				},
				payment_intent_data: {
					description: `Darowizna na grupę: "${group}"`,
					metadata: {
						group: group,
						support_type: 'one_time',
					},
				},
				line_items: [
					{
						price_data: {
							currency: 'pln',
							product_data: {
								name: `Jednorazowa darowizna na: "${group}"`,
								description: `Dziękujemy za wsparcie grupy "${group}"! Twoja darowizna pomoże nam w realizacji naszych celów.`,
							},
							unit_amount: amount,
						},
						quantity: 1,
					},
				],
				ui_mode: 'embedded',
				return_url: `${req.headers.get('origin')}/paymentResult?session_id={CHECKOUT_SESSION_ID}`,
			})
		}

		console.log(`💳 Sesja utworzona: ${session.id} (${monthlySupport ? 'subskrypcja' : 'jednorazowa'})`)

		return NextResponse.json({
			clientSecret: session.client_secret,
			sessionId: session.id,
		})
	} catch (error) {
		console.error('❌ Błąd w API:', error)
		return NextResponse.json({ error: (error as Error).message }, { status: 500 })
	}
}
