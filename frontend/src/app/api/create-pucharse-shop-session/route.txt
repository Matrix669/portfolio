import { NextRequest, NextResponse } from 'next/server'
import { stripe } from '@/lib/stripe'
import { FormData } from '@/app/schemas/ShopFormFinalizationSchema'


export async function POST(request: NextRequest) {
	try {
		const formData: FormData = await request.json()


		const cartItems = [
			{
				name: 'Produkt 1',
				amount: 1000,
				quantity: 2,
			},
		];

		// Calculate shipping costs based on your chosen method
		const getDeliveryCost = (deliveryMethod: string): number => {
			switch (deliveryMethod) {
				case 'Inpost Kurier':
				case 'DPD Kurier':
					return 1200 // 12.00 zł
				case 'Inpost Paczkomat':
					return 1000 // 10.00 zł
				default:
					return 1200
			}
		}

		const deliveryCost = getDeliveryCost(formData.delivery)

		// Prepare line items for Stripe
		const lineItems = [
			...cartItems.map(item => ({
				price_data: {
					currency: 'pln',
					product_data: {
						name: item.name,
					},
					unit_amount: item.amount,
				},
				quantity: item.quantity,
			})),
			{
				price_data: {
					currency: 'pln',
					product_data: {
						name: `Dostawa - ${formData.delivery}`,
					},
					unit_amount: deliveryCost,
				},
				quantity: 1,
			}
		]

		// Utwórz sesję Stripe Checkout
		const session = await stripe.checkout.sessions.create({
			ui_mode: 'embedded',
			line_items: lineItems,
			mode: 'payment',
			return_url: `${request.headers.get('origin')}/paymentResult?session_id={CHECKOUT_SESSION_ID}`,
			automatic_tax: { enabled: false },
			customer_email: formData.email,
			metadata: {
				customerData: JSON.stringify({
					firstName: formData.firstName,
					lastName: formData.lastName,
					email: formData.email,
					phoneNumber: formData.phoneNumber,
					address: {
						street: formData.street,
						houseNumber: formData.houseNumber,
						apartmentNumber: formData.apartmentNumber || '',
						town: formData.town,
						zipCode: formData.zipCode,
					},
					isCompany: formData.isCompany || false,
					NIP: formData.NIP || '',
					companyName: formData.companyName || '',
					deliveryToAnotherAddress: formData.deliveryToAnotherAddress,
					additionalComments: formData.additionalComments || '',
					paymentMethod: formData.paymentMethod,
					delivery: formData.delivery,
				}),
			},
			shipping_address_collection: {
				allowed_countries: ['PL'],
			},
			billing_address_collection: 'required',
		})

		return NextResponse.json({
			clientSecret: session.client_secret,
			sessionId: session.id,
		})
	} catch (error) {
		console.error('Błąd pod czasem tworzenia sesji płatności:', error)
		return NextResponse.json(
			{ error: 'Błąd pod czasem tworzenia sesji płatności' },
			{ status: 500 }
		)
	}
}
