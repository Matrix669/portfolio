'use server'

import { fetchProductStock } from './product-stock-helpers'

type VerifyProductStockError = 'OUT_OF_STOCK' | 'NOT_FOUND' | 'NETWORK_ERROR'

export type VerifyProductStockResult =
	| { success: true; availableQuantity: number }
	| { success: false; reason: VerifyProductStockError; availableQuantity?: number }

interface VerifyProductStockPayload {
	productId: string
	requestedQuantity: number
}

export async function verifyProductStockAction({
	productId,
	requestedQuantity,
}: VerifyProductStockPayload): Promise<VerifyProductStockResult> {
	try {
		const availableQuantity = await fetchProductStock(productId)

		if (typeof availableQuantity !== 'number') {
			return { success: false, reason: 'NOT_FOUND' }
		}

		if (requestedQuantity > availableQuantity) {
			return { success: false, reason: 'OUT_OF_STOCK', availableQuantity }
		}

		return { success: true, availableQuantity }
	} catch (error) {
		console.error('Błąd podczas weryfikacji stanu magazynowego produktu', error)
		return { success: false, reason: 'NETWORK_ERROR' }
	}
}
